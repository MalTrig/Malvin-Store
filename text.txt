<script>
const phone = "27754853690";

let cart = [];
let total = 0;

function scrollToShop() {
  document.getElementById("products").scrollIntoView({behavior:"smooth"});
}

function addToCart(name, price) {
  cart.push({name, price});
  total += price;

  const item = document.createElement("li");
  item.textContent = name + " - R" + price;
  document.getElementById("cart-items").appendChild(item);

  document.getElementById("total").textContent = "Total: R" + total;
}

function whatsappOrder(product, price) {
  const message = `Hello, I want to order:\n${product} - R${price}`;
  window.open(`https://wa.me/${phone}?text=${encodeURIComponent(message)}`);
}

function buyNow(product) {
  const message = `Quick Order:\n${product}`;
  window.open(`https://wa.me/${phone}?text=${encodeURIComponent(message)}`);
}

function submitOrder(e) {
  e.preventDefault();
  const inputs = document.querySelectorAll("#checkout input");

  let orderList = cart.map(i => i.name + " (R" + i.price + ")").join(", ");

  const message = `
NEW ORDER - MALVIN

Name: ${inputs[0].value}
Phone: ${inputs[1].value}
Location: ${inputs[2].value}
PEP PAXI: ${inputs[3].value}

Order: ${orderList}
Total: R${total}
`;

  window.open(`https://wa.me/${phone}?text=${encodeURIComponent(message)}`);
}
</script>
